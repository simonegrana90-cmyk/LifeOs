SQL da eseguire in Supabase → SQL Editor
create table public.lifeos_states (
  id uuid default gen_random_uuid() primary key,
  user_id uuid references auth.users(id) on delete cascade not null unique,
  state jsonb not null default '{}',
  updated_at timestamptz default now()
);
alter table public.lifeos_states enable row level security;
create policy "owner read"  on public.lifeos_states for select  using (auth.uid() = user_id);
create policy "owner write" on public.lifeos_states for insert  with check (auth.uid() = user_id);
create policy "owner update" on public.lifeos_states for update using (auth.uid() = user_id);
create policy "owner delete" on public.lifeos_states for delete using (auth.uid() = user_id);
